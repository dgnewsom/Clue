173 
174 
175  // Create board from string array
176  int x;
177  int z = -1;
178 
179  for (int row = boardStringArray.Length - 1; row >= 0; row--)
180  {
181  GameObject rowObject = new GameObject("Row - " + (boardStringArray.Length - row - 1));
182  rowObject.transform.parent = generalTiles.transform;
183  z++;
184  x = -1;
185  for (int col = 0; col < boardStringArray[row].Length; col++)
186  {
187  x++;
188  string[] square = boardStringArray[row][col].Split(':');
189  switch (square[0])
190  {
191  case "X":
192  CreateBoardTile(x, z, rowObject);
193  break;
194  case "S":
195  CreateStartTile(x, z, square[1], rowObject);
196  break;
197  case "E":
198  CreateRoomEntranceTile(x, z, square[1], square[2], rowObject);
199  break;
200  case "SC":
201  CreateShortcutTile(x, z, square[1], square[2]);
202  break;
203  case "R":
204  CreateRoom(x, z, square[1]);
205  break;
206  default:
207  break;
208  }
209  }
210 
211  //Delete row if empty
212  if (rowObject.transform.childCount.Equals(0))
213  {
214  GameObject.Destroy(rowObject);
215  }
